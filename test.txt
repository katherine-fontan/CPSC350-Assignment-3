#include "GenQueue.h"

GenQueue::GenQueue(){
  //default constructor
}
GenQueue::GenQueue(int maxSize){
  myQueue = new char[maxSize];
  mSize = maxSize;
  front = 0;
  rear = -1;
  numElements = 0;
}

GenQueue::~GenQueue(){
  //ill let you figure this out again
  delete myQueue;
}

void GenQueue::insert(char d){
  //add error checking
  myQueue[++rear] = d;
  ++numElements;
}

char GenQueue::remove(){
  char c  = '\0'; // initialize to null character

  c = myQueue[front];
  ++front;
  --numElements;
  return c;
}

char GenQueue:: peek(){
  return myQueue[front];
}

bool GenQueue::isFull(){
  return (numElements == mSize);
}

bool GenQueue:: isEmpty(){
  return (numElements == 0);
}

int GenQueue::getSize(){
  return numElements;
}
